4. bitsandbytes:	Enables 8-bit and 4-bit quantization.

		for page in doc:
			text = page.get_text('text').strip()
			if text:
				text_blocks.append(text)

def tokenize(example):
	tokens = tokenizer(example['text'].truncation=True, padding='max_length', max_length=512)
	tokens['labels'] = tokens['input_ids'].copy()
	return tokens

"""
In the above code segment, 
example["text"] = each entry in dataset object
	this looks something like
	{
		"text": mitrochondria is power house of the cell.
	}

tokenization tokens.
	truncation=True	# this means if text is longer than max_length, cut of it.
	padding="max_length" # if text shorter than 512 than add padding tokens.
	max_length=512 # fixed sequence length per training examples.

input_ids:[0, 1, 5, 7, 9, 15, ...] attention_mask: [0, 0, 1, 1, 1, 0]

tokens['labels'] = tokens['input_ids'].copy()
this is the key step for casual (unsupervised) language modeling. It means, the model should try to predict the next token of this same sequence.

so model learns: given previous words, predict the next word.

output:
	{
		"input_ids": []
		"attention_mask": [],
		"labels": []
	}
"""


# perform mapping
tokenized_data = dataset.map(tokenize, batched=True, remove_columns=["text"]) # here we remove the "text" from the output we only required, input_ids, attention_mask, labels.

# training arguments configuration
training_args = TrainingArguments(
	output_dir=f"{model_name}/domain-specific",	# in which output directory to save the models.
	overwrite_output_dir=True,			# allow overwrite, to avoid folder conflicts.
	num_train_epochs=2,				# training cycles,
	per_device_train_batch_size=2,			# batch per GPU, affects speed and memory
	save_steps=500,					# save frequency
	save_total_limit=2,				# keep last N models.
	logging_steps=50,				# log every X steps for progress monitoring
	learning_rate=2e-5,				# Weight update rate
	fp16=True,					# mixed precision, faster + less memory.
	report_to="none"				# logging destination
)

trainer = Trainer(
	model=model_name,
	args=training_args,
	train_dataset=tokenized_data)

trainer.train()
