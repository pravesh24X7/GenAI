OLD School Fine tuning technique:
In this Freeze some layers and finetune unfreeze layers.

import necessary lib's ...

Step 1: begin with loading model and settings up tokenizer
model_name=""
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForCasualLM.from_pretrained(model_name, device_map='auto')


Step 2: freeze everything
for param in model.parameters():
	param.requires_grad=False


Step 3: unfreeze last layers for training
for name, param in model.name_parameters():
	if any( f"layers.{i}." in name for i in range(20, 24) ):
		param.requires_grad=True
	if "lm_head" in name:
		param.requires_grad=True

Step 4: Verify the no. of training parameters
trainable = sum(p.numel() for p in model.parameters() if p.requires_grad)
total = sum(p.numel() for p in model.parameters())
print(f"Trainable: {trainable/total*100:.2f} of total parameters")

Step 5: load your dataset

dataset = load_dataset("json", data_files={"train": 'file.jsonl'})


Step 6: Define the tokenize function

def tokenize_fxn(example):
	tokens = tokenizer(example['text'], truncation=True, padding='max_length', max_lenght=512)
	tokens['labels'] = tokens['input_ids'].copy()
	return tokens

tokenized_data = dataset.map(tokenize_fxn, batched=True, remove_columns=["text"])

# Step 7: Data collector for casual LM
data_collector = DataCollectorForLanguageModeling(tokenizer=tokenizer, mlm=False)

# Step 8: Training arguments
os.environ['WANDB_DISABLED'] = "true"	# to disable WANDB
training_args = TrainingArguments(
	output_dir=f"{model_name}-domain-specific",
	overwrite_output_dir=True,
	num_train_epochs=2,
	per_device_train_batch_size=2,
	gradient_accumulation_steps=4,
	learning_rate=1e-4,
	fp16=True,
	logging_steps=20,
	save_steps=200,
	save_total_limit=2,
	report_to='non'
)

trainer = Trainer(training_args=training_args,
			model=model,
			train_dataset=tokenized_data,
			data_collector=data_collector)

trainer.train()

# Step 9: Save final model.
trainer.save_model("./final_model")
